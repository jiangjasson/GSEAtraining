The analysis starts from here.

The following function converts gene IDs to EntreZ gene ID:

convert_to_entrez_id = function(x) {
    if(is.matrix(x)) {
        map = cola:::guess_id_mapping(rownames(x))
        if(is.function(map)) {
            new_rn = map(rownames(x))
        } else {
            new_rn = map[rownames(x)]
        }
        l = is.na(new_rn)

        x = x[!l, , drop = FALSE]
        new_rn = new_rn[!l]

        x2 = do.call(rbind, tapply(1:nrow(x), new_rn, function(ind) {
            colMeans(x[ind, , drop = FALSE])
        }))
        return(x2)

    } else if(is.numeric(x)) {
        map = cola:::guess_id_mapping(names(x))
        x2 = s
        if(is.function(map)) {
            names(x2) = map(names(x))
        } else {
            names(x2) = map[names(x)]
        }
        x2 = x2[!is.na(names(x2))]
        x2 = tapply(x2, names(x2), mean)
        return(x2)
    } else {
        map = cola:::guess_id_mapping(x)
        if(is.function(map)) {
            x2 = map(x)
        } else {
            x2 = map[x]
        }
        x2 = x2[!is.na(x2)]
        x2 = unique(x2)
        return(x2)
    }
}

We apply t-test to every gene.

library(genefilter)
tdf = rowttests(expr, factor(condition))
tdf$fdr = p.adjust(tdf$p.value, "BH")
sum(tdf$fdr < 0.05)  # number of diff genes
## [1] 2

It seems there are not too many diff genes. We look at the distribution of the t-statistics:

plot(sort(tdf$statistic))

Instead of setting a cutoff for FDR, we set a cutoff for t-statistics, just to get enough number of diff genes.

sum(abs(tdf$statistic) > 2)
## [1] 487

First we perform ORA, using GO BP gene sets:

library(clusterProfiler)
library(org.Hs.eg.db)
l_sig = abs(tdf$statistic) > 2
tb_ora = enrichGO(gene  = rownames(expr)[l_sig],
                  OrgDb = org.Hs.eg.db,
                  ont   = "BP",
                  pAdjustMethod = "BH")
tb_ora = as.data.frame(tb_ora)

Then we perform GSEA, using t-statistics as gene scores. Note here the p-values are calculated by gene permutation.

s = tdf$statistic
names(s) = rownames(tdf)
s = sort(s, decreasing = TRUE)
tb_gsea = gseGO(geneList = s, 
                OrgDb = org.Hs.eg.db,
                ont   = "BP",
                pAdjustMethod = "BH")

Recall ORA is sensitive to selection of background genes and cutoff of diff genes. Next we first set background for ORA, which is the total genes in the matrix:

l_sig = abs(tdf$statistic) > 2
tb_ora_with_bg = enrichGO(gene = rownames(expr)[l_sig],

                  OrgDb = org.Hs.eg.db,
